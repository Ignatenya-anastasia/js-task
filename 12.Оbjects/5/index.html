Собери объект, где ключ — имя пользователя, а значение — массив его заказов. let users = [ { id: 1, name: "Anna" }, { id: 2, name: "Maks" } ]; let orders = [ { userId: 1, item: "Book" }, { userId: 2, item: "Phone" }, { userId: 1, item: "Pen" } ]; // Ожидается: { Anna: ["Book", "Pen"], Maks: ["Phone"] }
